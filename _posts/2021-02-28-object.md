```js
//오브젝트는 키와 밸류의 집합체이다.
//object = {key : value};

// const name = 'ellie';
// const age = 4;
// print(name, age);
// function print(name1, age){
//     console.log(name);
//     console.log(age);
// }
//이렇게 하게되면 주소나 풀네임 등 인자가 많아지면 추가해야 되는 것들이
//많아지고 logical하게 생각하기 어려워지게 된다.

//1.Literals and properties
const obj = {}; //'object literal'
const obj2 = new Object(); //'object constructor'

function print(person) {
    console.log(person.name);
    console.log(person.age);
}
const ellie = { name: 'ellie', age: 4 };
print(ellie);

ellie.hasjob = true;
console.log(ellie.hasjob);
//javascript에서는 뒤늦게 object에 property를 추가할수있다.
//이미 name과 age라는 오브젝트를 정의했지만 직업유무도 추가하고 싶다면 가능.
//이렇게 너무 동적으로 코딩하면 금방금방 하기에는 좋지만 나중에 유지보수가 힘들어지고
//생각지 못한 에러가 발생하게 될 수 있으니 가능하면 지양하는 것이 좋다.

delete ellie.hasjob;
console.log(ellie.hasjob);

//2.Computed properties(계산된 property)
//key는 항상 string 타입으로 가져와야한다.
console.log(ellie.name);
console.log(ellie['name']);
ellie['hasjob'] = true;
console.log(ellie.hasjob);

function printValue(obj, key) {
    // console.log(obj.key); 잘못된 예
    console.log(obj[key]);
}
printValue(ellie, 'name');

//3.Property value shorthand
const person1 = { name: 'bob', age: 2 };
const person2 = { name: 'steve', age: 3 };
const person3 = { name: 'dave', age: 4 };
const person4 = new Person('ellie', 30);
console.log(person4);

//4.Constructor function
function Person(name, age) {
    // this = {}; 생략
    this.name = name;
    this.age = age;
    // return this; 생략
}

//5.in operator : 해당하는 오브젝트 안에 키가 있는지 없는지 확인
console.log('name' in ellie);
console.log('age' in ellie);
console.log('random' in ellie);
console.log(ellie.random);

//6.for..in vs for..of (현업시 유용하게 사용하는 문법)
//for(key in obj)
console.clear();
for (key in ellie) {
    console.log(key);
}

//for(value of iterable)
const array = [1, 2, 4, 5];
// for(let i =0; i<array.length; i++){
//     console.log(array[i]);
// } 실용적이지 못한 코드
for (value of array) {
    console.log(value);
}

//7.cloning💖
//Object.assign(dest, [obj1, obj2, obj3...])
const user = { name: 'ellie', age: '20' };
const user2 = user;
user2.name = 'coder';
console.log(user);

//old way
const user3 = {};
for (key in user) {
    user3[key] = user[key];
}
console.clear();
console.log(user3);

const user4 = Object.assign({}, user);
console.log(user4);

//another example
const fruit1 = { color: 'red' };
const fruit2 = { color: 'blue', size: 'big' };
const mixed = Object.assign({}, fruit1, fruit2);
console.log(mixed.color); //blue -> 뒤에나오는 것 일수록 앞에 동일한 property가 있다면 값을 계속 덮어 씌운다.
console.log(mixed.size); //big
```
