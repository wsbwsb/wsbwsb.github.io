### 드림코딩 엘리 Javascript 강의 (7) - 유용한 10가지 배열함수들 Array APIs 총정리

```js
// Q1. make a string out of an array
// 배열로 string을 만들어라
{
    const fruits = ['apple', 'banana', 'orange'];
    const result = fruits.join();
    console.log(result);
}

// Q2. make an array out of a string
// string으로 배열을 만들어라
{
    const fruits = '🍎, 🥝, 🍌, 🍒';
    const result = fruits.split(',');
    console.log(result);
}

// Q3. make this array look like this: [5, 4, 3, 2, 1]
// [5, 4, 3, 2, 1] 이렇게 보이게 만들어라
{
    const array = [1, 2, 3, 4, 5];
    const result = array.reverse();
    console.log(result);
}

// Q4. make new array without the first two elements
// 첫번째부터 두개를 제외하고 새 배열을 만들어라
{
    const array = [1, 2, 3, 4, 5];
    const result = array.splice(0, 2);
    console.log(result);
    console.log(array);
}

class Student {
    constructor(name, age, enrolled, score) {
        this.name = name;
        this.age = age;
        this.enrolled = enrolled;
        this.score = score;
    }
}
const students = [
    new Student('A', 29, true, 45),
    new Student('B', 28, false, 80),
    new Student('C', 30, true, 90),
    new Student('D', 40, false, 66),
    new Student('E', 18, true, 88),
];

// Q5. find a student with the score 90
// 90점 받은 학생을 구하라
{
    const result = students.find((student) => student.score === 90);
    console.log(result);
}

// Q6. make an array of enrolled students
// 재학생을 배열로 만들어라.?
{
    const result = students.filter((student) => student.enrolled);
    console.log(result);
    //콜백함수가 true인 것만 출력
}

// Q7. make an array containing only the students' scores
// 학생의 성적만으로만 배열을 만들어라
// result should be: [45, 80, 90, 66, 88] //결과값은 반드시 이런 모양으로
{
    const result = students.map((student) => student.score * 2);
    console.log(result);
}
//map은 배열안에 있는 요소들을 원하는 함수를 이용해서 다른 방식의 데이터를 만들고 싶을때 이용하면 좋다

// Q8. check if there is a student with the score lower than 50
// 점수가 50점 미만인 학생 체크
{
    console.clear();
    const result = students.some((student) => student.score < 50);
    console.log(result);
    //some은 배열의 요소 중 콜백함수가 return이 true가 되는애가 있는지 없는지 확인

    const result2 = !students.every((student) => student.score >= 50);
    console.log(result2);
    //every는 배열에 있는 모든 요소들이 이 조건을 충족해야지만 true를 리턴
}

// Q9. compute students' average score
// 학생들의 평균점수 계산
{
    const result = students.reduce((prev, curr) => prev + curr.score, 0);
    console.log(result / students.length);
    //배열에 있는 모든 요소들의 값을 누적하는, 뭔가 함께 모아놓을 때 씀
}

// Q10. make a string containing all the scores
// 모든 점수가 포함된 string을 만들어라
// result should be: '45, 80, 90, 66, 88'
{
    const result = students
        .map((student) => student.score)
        .filter((score) => score >= 50)
        .join();
    console.log(result);
}

// Bonus! do Q10 sorted in ascending order
// Q10을 오름차순으로 정렬하라
// result should be: '45, 66, 80, 88, 90'
{
    const result = students
        .map((student) => student.score)
        .sort((a, b) => a - b)
        .join();
    console.log(result);
}
```
